using System;

public class Assignment_04
{
public static void Main()
{
	float Distance, Hour, Min, Sec, TimeSec, MPS, KPH, MPH ; // Declaring variables
	Console.Write("\nPlease enter Distance (in metres) :\n");
	Distance = Convert.ToSingle(Console.ReadLine());
	Console.Write("\nPlease enter Time (in hours) :\n");
	Hour = Convert.ToSingle(Console.ReadLine());
	Console.Write("\nPlease enter Time (in minutes) :\n");
	Min = Convert.ToSingle(Console.ReadLine());
	Console.Write("\nPlease enter Time (in seconds) :\n");
	Sec = Convert.ToSingle(Console.ReadLine());
	
	TimeSec = (Hour*3600) + (Min*60) + Sec ;
	MPS = Distance/TimeSec ;
	KPH = (Distance/1000.0f)/(TimeSec/3600.0f) ;
	MPH = KPH/1.609f ; 
	
	Console.WriteLine("Your Speed in metres.sec is {0}", MPS);
	Console.WriteLine("Your Speed in km/hour is {0}", KPH);
	Console.WriteLine("Your Speed in miles/hour is {0}", MPH);
}
}
